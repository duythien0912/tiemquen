Key code

Hai chỗ pin down nguyên tắc vàng: context không LLM trên buyer page, và SLA notify quan trọng nhất hệ thống.

buyer page · context rules (client/edge, không LLM)

// QR encode shop_id + batch_id; mọi biến thể đã compose sẵn
function pickVariant(qr, now, store) {
  const ui = cachedA2UI[qr.shop_id];        // không gọi LLM
  if (qr.batch.tag === 'office') ui.promote('group_order');
  if (now.hour >= 10 && now.hour <= 13)
      ui.reorderFirst('lunch_combo');  // reorder section
  const last = localStorage.get('last_order');
  if (last) ui.prepend(reorderCard(last)); // template + data cũ
  ui.hideItems(store.soldOut);              // hết món tự ẩn
  return ui;
}

# đơn tới mà quán không biết = khách mất vĩnh viễn
async def notify_seller(order):
    await fcm_push(order.seller, order)   # pilot: SMS là kênh chính
    await schedule(order.id, delay=120)   # chờ ack 2 phút

async def on_ack_timeout(order):
    if not order.acked:
        await send_sms(order.seller.phone, fmt(order))  # ~700đ
